Quick Start

# Install AI Context Nexus
git clone https://github.com/MCERQUA/AI-Context-Nexus
cd ai-context-nexus
./quickstart.sh

# Create your first context
from ai_context_nexus import ContextNexus

nexus = ContextNexus()

# Store context
context_id = nexus.create_context(
    content="Analysis of codebase architecture",
    context_type="technical",
    tags=["architecture", "analysis"]
)

# Semantic search
results = nexus.search("architecture patterns", limit=10)

# Multi-agent collaboration
agent1.submit_context("Initial analysis")
shared = nexus.get_shared_contexts()
agent2.process_with_context(shared)
